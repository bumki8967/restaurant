CREATE TABLE `user_info` (
  `USER_SEQ` int NOT NULL AUTO_INCREMENT,
  `USER_ID` varchar(50) NOT NULL COMMENT '아이디(=이메일)',
  `USER_PW` varchar(100) NOT NULL COMMENT '비밀번호',
  `NAME` varchar(10) NOT NULL COMMENT '이름',
  `TEL` varchar(20) DEFAULT NULL COMMENT '전화번호',
  `BIRTHDAY` varchar(20) DEFAULT NULL COMMENT '생년월일',
  `GENDER` varchar(10) DEFAULT NULL COMMENT '성별(남/여)',
  `POSTCODE` varchar(20) DEFAULT NULL COMMENT '우편번호',
  `ADDR1` varchar(50) DEFAULT NULL COMMENT '주소',
  `ADDR2` varchar(50) DEFAULT NULL COMMENT '상세주소',
  `REG_DATE` datetime DEFAULT NULL COMMENT '가입일자',
  `EDIT_DATE` datetime DEFAULT NULL COMMENT '수정일자',
  `USER_TYPE` varchar(10) DEFAULT NULL COMMENT '구분(일반/관리자)',
  `LOGIN_TYPE` varchar(10) DEFAULT NULL COMMENT 'SNS 로그인 구분',
  `ADD_ITEM1` varchar(255) DEFAULT NULL,
  `ADD_ITEM2` varchar(255) DEFAULT NULL,
  `ADD_ITEM3` varchar(2555) DEFAULT NULL,
  PRIMARY KEY (`USER_SEQ`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb3 COMMENT='회원정보 테이블'
